- 四维，不优化版
```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 55;

int n, m;
int g[N][N];
int f[N][N][N][N];

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i ++ ) 
        for (int j = 1; j <= m; j ++ )
            cin >> g[i][j];
    

    for (int i1 = 1; i1 <= n; i1 ++ )
        for (int i2 = 1; i2 <= n; i2 ++ )
            for (int j1 = 1; j1 <= m; j1 ++ )
                for (int j2 = 1; j2 <= m; j2 ++ )
                {
                    int t = g[i1][j1];
                    if (i1 != i2 || j1 != j2) t += g[i2][j2];
                    
                    int& x = f[i1][i2][j1][j2];
                    x = max(x, f[i1 - 1][i2 - 1][j1][j2] + t);
                    x = max(x, f[i1][i2 - 1][j1 - 1][j2] + t);
                    x = max(x, f[i1 - 1][i2][j1][j2 - 1] + t);
                    x = max(x, f[i1][i2][j1 - 1][j2 - 1] + t);
                }
            
    cout << f[n][n][m][m] << endl;
    return 0;
}
```

- 三维，优化版
```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 55;

int n, m;
int g[N][N];
int f[2 * N][N][N];

int main()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i ++ ) 
        for (int j = 1; j <= m; j ++ )
            cin >> g[i][j];
    
    for (int k = 2; k <= n + m; k ++ )
        for (int i1 = 1; i1 <= n; i1 ++ )
            for (int i2 = 1; i2 <= n; i2 ++ )
            {
                int j1 = k - i1, j2 = k - i2;
                if (j1 < 1 || j1 > m || j2 < 1 || j2 > m) continue;
                
                int t = g[i1][j1];
                if (i1 != i2) t += g[i2][j2];
                
                int& x = f[k][i1][i2];
                x = max(x, f[k - 1][i1 - 1][i2] + t);
                x = max(x, f[k - 1][i1][i2 - 1] + t);
                x = max(x, f[k - 1][i1 - 1][i2 - 1] + t);
                x = max(x, f[k - 1][i1][i2] + t);
            }
            
    cout << f[n + m][n][n] << endl;
    return 0;
}
```
