```c++
#include <iostream>
#include <queue>
#include <cstring>

using namespace std;

typedef pair<int, int> PII;

const int N = 1010;

int n;
int g[N][N];
int dist[N][N];
PII pre[N][N];

void bfs()
{
    int dx[] = {-1, 0, 1, 0}, dy[] = {0, 1, 0, -1};
    memset(dist, -1, sizeof dist);
    dist[n - 1][n - 1] = 0;
    queue<PII> q;
    q.push({n - 1, n - 1});
    
    while (q.size())
    {
        auto t = q.front();
        q.pop();
        
        for (int i = 0; i < 4; i ++ )
        {
            int a = t.first + dx[i], b = t.second + dy[i];
            if (a < 0 || a >= n || b < 0 || b >= n) continue;
            if (dist[a][b] != -1 || g[a][b]) continue;
            
            dist[a][b] = dist[t.first][t.second] + 1;
            q.push({a, b});
            pre[a][b] = t;
        }
    }
}

int main()
{
    cin >> n;
    for (int i = 0; i < n ; i ++ )
        for (int j = 0; j < n; j ++ )
            cin >> g[i][j];
    
    bfs();
    
    PII end = {0, 0};
    while (true)
    {
        cout << end.first << ' ' << end.second << endl;
        if (end.first == (n - 1) && end.second == (n - 1)) break;
        end = pre[end.first][end.second];
    }
    
    return 0;
}
```

```c++
#include <bits/stdc++.h>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;

const int N = 1010, M = N * N;

int n;
int g[N][N];
PII q[M];
PII pre[N][N];

void bfs(int sx, int sy)
{
    int dx[4] = {0, -1, 0, 1}, dy[4] = {-1, 0, 1, 0};
        
    int hh = 0, tt = -1;
    q[ ++ tt] = {sx, sy};
    
    memset(pre, -1, sizeof pre);
    pre[sx][sy] = {0, 0};

    while (hh <= tt)
    {
        auto t = q[hh ++ ];

        for (int i = 0; i < 4; i ++ )
        {
            int a = t.x + dx[i], b = t.y + dy[i];
            if (a < 0 || a >= n || b < 0 || b >= n) continue;
            if (pre[a][b].x != -1 || g[a][b]) continue;
            
            q[ ++ tt] = {a, b};
            pre[a][b] = t;
        }
    }
    
    PII end = {0, 0};
    while (true)
    {
        //三句话的顺序：
        //输出、判断、迭代
        //如果迭代完了立刻判断，终点还未来得及输出就break了
        //输出完了就判断是否需要break
        cout << end.x << ' ' << end.y << endl;
        if (end.x == n - 1 && end.y == n - 1) break;
        end = pre[end.x][end.y];
    }
}

int main()
{
    cin >> n;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < n; j ++ )
            cin >> g[i][j];
            
    bfs(n - 1, n - 1);
    
    return 0;
}
```
