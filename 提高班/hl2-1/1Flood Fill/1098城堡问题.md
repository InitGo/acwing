```c++
#include <iostream>
#include <queue>

using namespace std;

const int N = 55;

int n, m;
int g[N][N];
bool st[N][N];
int dx[] = {0, -1, 0, 1}, dy[] = {-1, 0, 1, 0};
int cnt, maxn;

void bfs(int x, int y)
{
    int area = 0;
    st[x][y] = true;
    
    queue<pair<int, int>> q;
    q.push({x, y});
    
    while (q.size()) 
    {
        auto t = q.front();
        area ++ ;
        q.pop();
        
        for (int i = 0; i < 4; i ++ )
        {
            if (g[t.first][t.second] >> i & 1) continue;
            int a = t.first + dx[i], b = t.second + dy[i];
            if (a < 0 || a >= n || b < 0 || b >= m) continue;
            if (st[a][b]) continue;
            
            q.push({a, b});
            st[a][b] = true;
        } 
    }
    
    maxn = max(maxn, area);
}

int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ ) 
            cin >> g[i][j];
    
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            if (!st[i][j]) bfs(i, j), cnt ++ ;

    cout << cnt << endl;
    cout << maxn << endl;
    
    return 0;
}
```

```c++
#include <cstring>
#include <iostream>
#include <algorithm>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;

const int N = 55, M = N * N;

int n, m;
int g[N][N];
PII q[M];
bool st[N][N];

int bfs(int sx, int sy)
{
    int dx[4] = {0, -1, 0, 1}, dy[4] = {-1, 0, 1, 0};

    int hh = 0, tt = 0;
    int area = 0;

    q[0] = {sx, sy};
    st[sx][sy] = true;

    while (hh <= tt)
    {
        PII t = q[hh ++ ];
        area ++ ; //在出队列的时候计算面积

        for (int i = 0; i < 4; i ++ ) //遍历0123四个方向的点
        {
            int a = t.x + dx[i], b = t.y + dy[i];
            if (a < 0 || a >= n || b < 0 || b >= m) continue;
            if (st[a][b]) continue;
            if (g[t.x][t.y] >> i & 1) continue; //该点是墙 continue

            q[ ++ tt] = {a, b};
            st[a][b] = true;
        }
    }

    return area;
}

int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            cin >> g[i][j];

    int cnt = 0, area = 0;
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            if (!st[i][j])
            {
                area = max(area, bfs(i, j));
                cnt ++ ;
            }

    cout << cnt << endl;
    cout << area << endl;

    return 0;
}
```
