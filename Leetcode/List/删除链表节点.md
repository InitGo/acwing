### 给出具体的节点，删除这个节点

思路：一个取巧的方式，让当前需要被删除的节点的值等于下一个节点的值，然后删除下一个节点，效果上等价于删除了当前节点
避免了删除节点需要前驱节点的问题
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    void deleteNode(ListNode* node) {
        node->val = node->next->val;
        node->next = node->next->next;
    }
};
```


### 给出一个val，将值为这个val的节点删除
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteNode(ListNode* head, int val) {
        if (head->val == val) return head->next;

        auto cur = head;
        while (cur->next)
        {
            if (cur->next->val == val) 
            {
                cur->next = cur->next->next;
                break;
            }
            else cur = cur->next;   
        }
        return head;
    }
};

```
