### 算法1 小学数奥

```c++
class Solution {
public:
    int cuttingRope(int n) {
        if (n <= 3) return 1 * (n - 1);
        int res = 1;
        if (n % 3 == 1) res *= 4, n -= 4;
        if (n % 3 == 2) res *= 2, n -= 2;
        while (n) res *= 3, n -= 3;
        return res;
    }
};
```

### 算法2 DP
- 2 <= n <= 57
```c++
class Solution {
public:
    int cuttingRope(int n) {
        int f[60];
        memset(f, 0, sizeof f);

        for (int i = 1; i <= n; i ++ )
        {
            f[i] = max(i - 1, 1);
            for (int k = 1; k < i; k ++ )
                f[i] = max(f[i], max(f[k] * (i - k), (i - k) * k));
        }
        return f[n];
    }
};
```

- 2 <= n <= 100
```c++
class Solution {
public:
    int cuttingRope(int n) {
        if (n <= 3) return 1 * (n - 1);
        const int mod = 1e9 + 7;
        long long res = 1;
        if (n % 3 == 1) res *= 4, n -= 4;
        if (n % 3 == 2) res *= 2, n -= 2;
        while (n) res = res * 3 % mod, n -= 3;
        return res;
    }
};
```
