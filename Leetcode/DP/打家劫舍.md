- 不能偷相邻
```c++
class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        int f[n + 2][2];
        memset(f, -0x3f, sizeof f);
        f[0][0] = 0;
        for (int i = 1; i <= n; i ++ ) {
            f[i][1] = f[i - 1][0] + nums[i - 1];
            f[i][0] = max(f[i - 1][0], f[i - 1][1]);
        }

        return max(f[n][0], f[n][1]);
    }
};
```
