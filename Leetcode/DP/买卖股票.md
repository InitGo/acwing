- 只能买一次
```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int ans = 0, min_price = INT_MAX;
        for (int x : prices) {
            ans = max(ans, x - min_price);
            min_price = min(x, min_price);
        }

        return ans;
    }
};
```

- 买任意次
```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        int f[n + 2][2];
        f[0][0] = 0; //状态机入口
        f[0][1] = INT_MIN;
        for (int i = 1; i <= n; i ++ ) {
            f[i][0] = max(f[i - 1][0], f[i -1][1] + prices[i - 1]);
            f[i][1] = max(f[i - 1][1], f[i - 1][0] - prices[i - 1]);
        }
        return f[n][0];
    }
};
```

- 最多交易两次
```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        int f[n + 2][3][2];
        memset(f, -0x3f, sizeof f);
        for (int i = 0; i <= n; i ++ ) f[i][0][0] = 0;

        for (int i = 1; i <= n; i ++ ) 
            for (int j = 1; j <= 2; j ++ ) {
                f[i][j][0] = max(f[i - 1][j][0], f[i - 1][j][1] + prices[i - 1]);
                f[i][j][1] = max(f[i - 1][j][1], f[i - 1][j - 1][0] - prices[i - 1]);
            }
        
        int ans = 0;
        for (int j = 1; j <= 2; j ++ ) ans = max(ans, f[n][j][0]);
        return ans;
    }
};
```

- 含一天冷冻期
```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        int f[n + 2][3];
        memset(f, -0x3f, sizeof f);
        f[0][2] = 0;
        for (int i = 1; i <= n; i ++ ) {
            f[i][0] = max(f[i-1][0], f[i-1][2] - prices[i-1]);
            f[i][1] = f[i - 1][0] + prices[i - 1];
            f[i][2] = max(f[i - 1][1], f[i - 1][2]);
        }

        return max(f[n][1], f[n][2]);
    }
};
```
